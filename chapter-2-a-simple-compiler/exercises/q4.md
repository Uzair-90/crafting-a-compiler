### Question No. 4

Write the recursive-descent parsing procedures for all nonterminals in ![image](../../images/fig-2.1.png)

### Solution

```C
procedure MATCH(expected)

    if lookahead.type = expected
    then
        lookahead ← SCANNER()

    else
        call ERROR()

end



procedure PROG()

    call DCLS()
    call STMTS()
    MATCH($)

end



procedure DCLS()

    if lookahead.type = floatdcl
       or lookahead.type = intdcl
    then
        call DCL()
        call DCLS()

    else
        ; λ production (do nothing)

end



procedure DCL()

    if lookahead.type = floatdcl
    then
        MATCH(floatdcl)
        MATCH(id)

    else if lookahead.type = intdcl
    then
        MATCH(intdcl)
        MATCH(id)

    else
        call ERROR()

end



procedure STMTS()

    if lookahead.type = id
       or lookahead.type = print
    then
        call STMT()
        call STMTS()

    else
        ; λ production (do nothing)

end



procedure STMT()

    if lookahead.type = id
    then
        MATCH(id)
        MATCH(assign)
        call VAL()
        call EXPR()

    else if lookahead.type = print
    then
        MATCH(print)
        MATCH(id)

    else
        call ERROR()

end



procedure EXPR()

    if lookahead.type = plus
    then
        MATCH(plus)
        call VAL()
        call EXPR()

    else if lookahead.type = minus
    then
        MATCH(minus)
        call VAL()
        call EXPR()

    else
        ; λ production (do nothing)

end



procedure VAL()

    if lookahead.type = id
    then
        MATCH(id)

    else if lookahead.type = inum
    then
        MATCH(inum)

    else if lookahead.type = fnum
    then
        MATCH(fnum)

    else
        call ERROR()

end

```
